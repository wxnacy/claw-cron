```python
async def _capture_feishu_openid(alias: str) -> None:
    """Capture OpenID from Feishu WebSocket.
    
    Args:
        alias: Contact alias to save the captured openid.
    """
    import lark_oapi as lark
    
    from claw_cron.feishu.events import parse_feishu_message
    
    # Load config
    config = load_config()
    feishu_config = config.get("channels", {}).get("feishu", {})
    app_id = feishu_config.get("app_id")
    app_secret = feishu_config.get("app_secret")
    
    if not app_id or not app_secret:
        console.print("[red]Error: Feishu not configured.[/red]")
        console.print("[dim]Run 'claw-cron channels add' first.[/dim]")
        raise SystemExit(1)
    
    captured_openid: str | None = None
    
    def on_message_received(data: lark.im.v1.P2ImMessageReceiveV1) -> None:
        """Handle message received event from Feishu WebSocket."""
        nonlocal captured_openid
        # Extract open_id from event (v2.0 structure)
        event_data = {
            "sender": {
                "sender_id": {
                    "open_id": data.event.sender.sender_id.open_id
                }
            },
            "message": {
                "content": data.event.message.content,
                "message_id": data.event.message.message_id,
                "chat_id": data.event.message.chat_id,
                "create_time": data.event.message.create_time
            }
        }
        message = parse_feishu_message(event_data)
        captured_openid = message.openid
        console.print(f"\n[green]✓ OpenID captured: [bold]{message.openid}[/bold][/green]")
        console.print(f"[dim]Message content: {message.content}[/dim]")
    
    # Build event handler
    event_handler = (
        lark.EventDispatcherHandler.builder("", "")
        .register_p2_im_message_receive_v1(on_message_received)
        .build()
    )
    
    # Create WebSocket client
    ws_client = lark.ws.Client(
        app_id,
        app_secret,
        event_handler=event_handler,
        log_level=lark.LogLevel.INFO
    )
    
    console.print("\n[bold]Waiting for message...[/bold]")
    console.print("[dim]Send any message to your Feishu bot to capture your openid.[/dim]")
    console.print("[dim]Press Ctrl+C to cancel.[/dim]\n")
    
    try:
        async def wait_for_capture() -> None:
            """Wait until openid is captured."""
            while not captured_openid:
                await asyncio.sleep(0.5)
        
        # Start WebSocket connection
        await asyncio.gather(
            ws_client.start(),  # This blocks
            wait_for_capture()
        )
    except KeyboardInterrupt:
        console.print("\n[yellow]Cancelled.[/yellow]")
        return
    
    # Save contact
    if captured_openid:
        contact = Contact(
            openid=captured_openid,
            channel="feishu",
            alias=alias,
            created=datetime.now().isoformat(),
        )
        save_contact(contact)
        console.print(f"\n[green]✓ Contact saved as '[bold]{alias}[/bold]'[/green]")
        console.print(f"[dim]You can now use 'claw-cron remind --recipient {alias}'[/dim]")
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User input → CLI | Untrusted input via click prompts (app_id, app_secret, alias) |
| CLI → Config file | Secrets stored in config.yaml (file system boundary) |
| CLI → Feishu API | Network boundary for credential validation |
| WebSocket → CLI | External event data from Feishu platform |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-12-06 | Information Disclosure | App secret in logs | mitigate | Use hide_input=True for secret prompt, never log app_secret |
| T-12-07 | Information Disclosure | Config file secrets | accept | User's local config, protected by file permissions (standard practice) |
| T-12-08 | Tampering | Alias injection | mitigate | Validate alias format (alphanumeric, underscore, no special chars) |
| T-12-09 | Spoofing | Fake WebSocket events | accept | Events come from authenticated Feishu WebSocket connection |
| T-12-10 | Denial of Service | WebSocket blocking | accept | User can Ctrl+C to cancel, no unbounded wait |
</threat_model>
